  Agentica,
  AgenticaHistory,
  IAgenticaHistoryJson,
} from "@agentica/core";
import OpenAI from "openai";

export const main = async (id: string): Promise<void> => {
  const agent = new Agentica({
    vendor: {
      api: new OpenAI({ apiKey: process.env.OPENAI_API_KEY }),
      model: process.env.OPENAI_MODEL ?? "gpt-4o-mini",
    },
    controllers: [...],
    histories: (await getHistories(id)) satisfies IAgenticaHistoryJson[],
  });
  const histories: AgenticaHistory[] = await agent.conversate(
    "I wanna buy MacBook Pro",
  );
  await archiveHistories(id, histories.map((history) => history.toJSON()));
};

`Agentica.conversate()` returns the histories created during the turn. Store `history.toJSON()` results when you need to resume a session, then pass those JSON records to `IAgenticaProps.histories` when constructing the next `Agentica`.

Histories are also useful for trimming context. A common compacting strategy is to keep user/assistant/system messages and `describe` summaries, while removing old `select`, `cancel`, and `execute` histories after their results have already been described.

  "I wanna buy MacBook Pro",
);

  if (history.type === "userMessage")
    console.log(history.contents);
  if (history.type === "assistantMessage")
    console.log(history.text);
  if (history.type === "systemMessage")
